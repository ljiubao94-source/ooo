      const currentUser = "<?!= currentUser; ?>";

      document.addEventListener("DOMContentLoaded", function() {
        const today = new Date();
        const y = today.getFullYear();

      function toggleFuelSection() {
        const mainBtn = document.getElementById("fuelMainBtn");
        const srcArea = document.getElementById("fuelSourceArea");
        const typeArea = document.getElementById("fuelTypeArea");
        const volArea = document.getElementById("fuelVolumeArea");
        const currentSource = document.getElementById("fuelSource").value;

        if (!currentSource) {
          mainBtn.style.background = "#9b59b6"; mainBtn.style.color = "#fff"; mainBtn.innerText = "🟪 給油・燃料入力中...";
          srcArea.style.display = "block";
        } else {
          document.getElementById("fuelSource").value = "";
          document.getElementById("fuelType").value = "";
          document.getElementById("fuelVolume").value = "";
          mainBtn.style.background = "#fff"; mainBtn.style.color = "#333"; mainBtn.innerText = "⬜ 給油・燃料入力";
          srcArea.style.display = "none";
          typeArea.style.display = "none";
          volArea.style.display = "none";
          resetSourceButtons();
        }
      }

      function selectFuelSource(source) {
        document.getElementById("fuelSource").value = source;
        const btnDHO = document.getElementById("f-src-DHO");
        const btnKig = document.getElementById("f-src-kig");

        if(source === 'DHO') {
          btnDHO.style.background = "#9b59b6"; btnDHO.style.color = "#fff";
          btnKig.style.background = "#fff"; btnKig.style.color = "#333";
        } else if(source === 'キグナス') {
          btnKig.style.background = "#9b59b6"; btnKig.style.color = "#fff";
          btnDHO.style.background = "#fff"; btnDHO.style.color = "#333";
        }
        updateFuelMainBtnText();
        buildFuelTypeButtons();
      }

      function resetSourceButtons() {
        document.getElementById("f-src-DHO").style.background = "#fff"; document.getElementById("f-src-DHO").style.color = "#333";
        document.getElementById("f-src-kig").style.background = "#fff"; document.getElementById("f-src-kig").style.color = "#333";
      }

      function buildFuelTypeButtons(selectedType = "") {
        const source = document.getElementById("fuelSource").value;
        const container = document.getElementById("fuelTypeButtons");
        const typeArea = document.getElementById("fuelTypeArea");
        container.innerHTML = "";
        
        let types = [];
        if (source === "DHO") types = ["軽油", "アドブルー"];
        if (source === "キグナス") types = ["軽油", "ガソリン", "混合"];

        if(types.length > 0) {
          typeArea.style.display = "block";
          types.forEach(type => {
            container.innerHTML += `<button type="button" class="fuel-sub-btn" id="f-type-${type}" onclick="selectFuelType('${type}')">${type}</button>`;
          });
          if(selectedType) selectFuelType(selectedType);
        }
      }

      function selectFuelType(type) {
        document.getElementById("fuelType").value = type;
        const buttons = document.getElementById("fuelTypeButtons").getElementsByTagName("button");
        
        for(let btn of buttons) {
          if(btn.id === `f-type-${type}`) {
            btn.style.background = "#28a745"; btn.style.color = "#fff";
          } else {
            btn.style.background = "#fff"; btn.style.color = "#333";
          }
        }
        updateFuelMainBtnText();
        document.getElementById("fuelVolumeArea").style.display = "block";
      }

      function updateFuelMainBtnText() {
        const src = document.getElementById("fuelSource").value;
        const type = document.getElementById("fuelType").value;
        const vol = document.getElementById("fuelVolume").value;
        const mainBtn = document.getElementById("fuelMainBtn");

        if(src && type && vol) {
          mainBtn.innerText = `🟪 ${src} / ${type} (${vol}L)`;
        } else if(src && type) {
          mainBtn.innerText = `🟪 ${src} / ${type}`;
        } else if(src) {
          mainBtn.innerText = `🟪 ${src} を選択中...`;
        }
      }

      function setFuelVolume(vol) {
        document.getElementById("fuelVolume").value = parseFloat(vol).toFixed(1);
        updateFuelMainBtnText();
      }

      function adjustFuelVolume(amount) {
        const input = document.getElementById("fuelVolume");
        let current = parseFloat(input.value) || 0;
        current += amount;
        if(current < 0) current = 0;
        input.value = current.toFixed(1);
        updateFuelMainBtnText();
      }

      document.getElementById("fuelVolume").addEventListener("input", updateFuelMainBtnText);

      function toggleViewArea() {
        const area = document.getElementById("viewArea");
        area.style.display = (area.style.display === "block") ? "none" : "block";
      }

      function fetchWorkerReport() {
        const workerName = document.getElementById("viewWorkerSelect").value;
        const targetMonth = document.getElementById("viewMonthSelect").value;
        const resultDiv = document.getElementById("reportResult");
        
        if(!workerName) { resultDiv.innerHTML = ""; return; }
        resultDiv.innerHTML = "<p style='font-size:24px; color:#666;'>データを読み込み中...</p>";

        google.script.run
          .withSuccessHandler(function(records) {
            if(!records || records.length === 0) {
              resultDiv.innerHTML = "<p style='font-size:24px; color:red;'>選択された月のデータが見つかりません</p>";
              return;
            }
            let html = "<table class='report-table'><tr><th>日</th><th>現場</th><th>人工</th><th>残業</th></tr>";
            records.forEach(r => {
              html += "<tr><td>" + r.day + "</td><td style='text-align:left;'>" + r.site + "</td><td>" + r.count + "</td><td>" + r.overtime + "</td></tr>";
            });
            html += "</table>";
            resultDiv.innerHTML = html;
          })
          .getWorkerMonthlyReport(workerName, targetMonth);
      }

      function toggleWorker(idx){
        const el = document.getElementById("w-btn-" + idx);
        const countDiv = document.getElementById("w-count-" + idx);
        const overDiv = document.getElementById("w-over-" + idx);
        const name = el.dataset.worker;
        const currentStatus = el.dataset.status;
        const isSpec = el.dataset.isspec === "true";

        if (currentStatus === "none") {
          el.dataset.status = "normal"; el.dataset.hours = ""; el.style.background = "#28a745"; el.style.color = "#fff";
          if(isSpec) {
            el.innerText = "🟩 " + name + " (1人)"; if(countDiv) countDiv.style.display = "block"; resetCountButtons(idx, 1);
          } else {
            el.innerText = "🟩 " + name; if(countDiv) countDiv.style.display = "none";
          }
          if(overDiv) overDiv.style.display = "none";
        } else if (currentStatus === "normal") {
          el.dataset.status = "half"; el.dataset.hours = ""; el.style.background = "#ffc107"; el.style.color = "#000"; el.innerText = "🟨 " + name + "（半日）";
          if(countDiv) countDiv.style.display = "none"; if(overDiv) overDiv.style.display = "none";
        } else if (currentStatus === "half") {
          el.dataset.status = "over"; el.dataset.hours = "1"; el.style.background = "#dc3545"; el.style.color = "#fff";
          if(isSpec) {
            const currentCount = el.dataset.count || "1"; el.innerText = "🟥 " + name + " (" + currentCount + "人 / 残業 1h)";
          } else {
            el.innerText = "🟥 " + name + "（残業 1h）";
          }
          if(countDiv) countDiv.style.display = "none"; if(overDiv) overDiv.style.display = "block";
        } else {
          el.dataset.status = "none"; el.dataset.hours = ""; el.dataset.count = "1"; el.style.background = "#fff"; el.style.color = "#000"; el.innerText = "⬜ " + name;
          if(countDiv) countDiv.style.display = "none"; if(overDiv) overDiv.style.display = "none";
        }
      }

      function setCount(idx, count) {
        const el = document.getElementById("w-btn-" + idx); const name = el.dataset.worker;
        el.dataset.count = count; el.innerText = "🟩 " + name + " (" + count + "人)";
        resetCountButtons(idx, count);
      }

      function resetCountButtons(idx, activeCount) {
        const countDiv = document.getElementById("w-count-" + idx); if(!countDiv) return;
        const btns = countDiv.getElementsByTagName("button");
        for(let i=0; i<btns.length; i++) {
          btns[i].style.background = ((i+1) === parseInt(activeCount)) ? "#28a745" : "#fff";
          btns[i].style.color = ((i+1) === parseInt(activeCount)) ? "#fff" : "#333";
        }
      }

      function save() {
        const site = document.getElementById("site").value;
        const workDate = document.getElementById("workDate").value;
        const concVol = document.getElementById("concreteVolume").value; 
        const fuelSrc = document.getElementById("fuelSource").value;
        const fuelType = document.getElementById("fuelType").value;
        const fuelVol = document.getElementById("fuelVolume").value;

        if (!site) { alert("現場を選択してください"); return; }
        if (!workDate) { alert("登録日を入力してください"); return; }

        let workers = [];
        document.querySelectorAll('.worker-btn').forEach(btn => {
          const status = btn.dataset.status;
          if (status !== "none") {
            workers.push({ name: btn.dataset.worker, status: status, hours: btn.dataset.hours, count: btn.dataset.count || "1" });
          }
        });

        let equipment = [];
        document.querySelectorAll('.equipment-checkbox:checked').forEach(cb => {
          equipment.push({ company: cb.getAttribute('data-company'), machine: cb.value });
        });

        const mainBtn = document.querySelector('.btn-main');
        mainBtn.innerText = "登録中..."; mainBtn.disabled = true;

        google.script.run
          .withSuccessHandler(function(){
            document.getElementById("successModal").style.display = "flex";
          })
          .saveData(site, workers, equipment, workDate, concVol, fuelSrc, fuelType, fuelVol);
      }

      function closeSuccessModal() {
        document.getElementById("successModal").style.display = "none";
        location.reload();
      }

      function loadLastData() {
        const site = document.getElementById("site").value; if (!site) { alert("現場を選択してください"); return; }
        google.script.run
          .withSuccessHandler(function(result){
            if (!result) return;
            document.getElementById("concreteVolume").value = result.concreteVolume || "";
            
            if(result.fuelSource) {
              document.getElementById("fuelMainBtn").style.background = "#9b59b6";
              document.getElementById("fuelMainBtn").style.color = "#fff";
              document.getElementById("fuelSourceArea").style.display = "block";
              selectFuelSource(result.fuelSource);
              if(result.fuelType) {
                selectFuelType(result.fuelType);
              }
              if(result.fuelVolume) {
                document.getElementById("fuelVolume").value = result.fuelVolume;
                updateFuelMainBtnText();
              }
            }

            document.querySelectorAll('.worker-btn').forEach((btn, idx) => {
              const name = btn.dataset.worker; const overDiv = document.getElementById("w-over-" + idx); const countDiv = document.getElementById("w-count-" + idx);
              const foundWorker = result.workers.find(w => w.name === name); const isSpec = btn.dataset.isspec === "true";

              if (foundWorker) {
                const status = foundWorker.status; btn.dataset.status = status; btn.dataset.count = foundWorker.count || "1";
                if (status === "half") {
                  btn.style.background = "#ffc107"; btn.style.color = "#000"; btn.innerText = "🟨 " + name + "（半日）";
                  if(overDiv) overDiv.style.display = "none"; if(countDiv) countDiv.style.display = "none";
                } else if (status === "over") {
                  const hrs = foundWorker.hours || "1"; btn.dataset.hours = hrs; btn.style.background = "#dc3545"; btn.style.color = "#fff";
                  btn.innerText = isSpec ? "🟥 " + name + " (" + btn.dataset.count + "人 / 残業 " + hrs + "h)" : "🟥 " + name + "（残業 " + hrs + "h）";
                  if(overDiv) overDiv.style.display = "block"; if(countDiv) countDiv.style.display = "none";
                } else {
                  btn.style.background = "#28a745"; btn.style.color = "#fff";
                  btn.innerText = isSpec ? "🟩 " + name + " (" + btn.dataset.count + "人)" : "🟩 " + name;
                  if(countDiv) countDiv.style.display = isSpec ? "block" : "none"; if(isSpec) resetCountButtons(idx, btn.dataset.count);
                  if(overDiv) overDiv.style.display = "none";
                }
              } else {
                btn.dataset.status = "none"; btn.dataset.hours = ""; btn.dataset.count = "1"; btn.style.background = "#fff"; btn.style.color = "#000"; btn.innerText = "⬜ " + name;
                if(overDiv) overDiv.style.display = "none"; if(countDiv) countDiv.style.display = "none";
              }
            });

            document.querySelectorAll('.equipment-checkbox').forEach(cb => {
              const comp = cb.getAttribute('data-company'); const mach = cb.value;
              const isChecked = result.equipment.some(e => e.company === comp && e.machine === mach); cb.checked = isChecked;
